canvas绘图

            移动画笔绘图

            var myCanvas =document.getElementbyId('myCanvas');
            var cxt = myCanvas.getContext('2d');

            基本绘图步骤:
            cxt.lineWidth = 8;
            cxt.fillStyle = 'red'; // 填充颜色
            cxt.strokeStyle = 'red'; // 路径颜色
            cxt.beginPath();    //定义起始从画布0,0 开始计算坐标
            cxt.moveTo(x,y);
            cxt.bezierCurveTo(x1,y1,x2,y2,x,y); //贝塞尔曲线
            cxt.fill();  // 填充
            cxt.stroke(); // 绘制路径
